25  """Deprecated module which sets the default GLib main context as the mainloop 
26  implementation within D-Bus, as a side-effect of being imported! 
27   
28  This API is highly non-obvious, so instead of importing this module, 
29  new programs which don't need pre-0.80 compatibility should use this 
30  equivalent code:: 
31   
32      from dbus.mainloop.glib import DBusGMainLoop 
33      DBusGMainLoop(set_as_default=True) 
34  """ 
35  __docformat__ = 'restructuredtext' 
36   
37  from dbus.mainloop.glib import DBusGMainLoop, threads_init 
38  from warnings import warn as _warn 
39   
40  init_threads = threads_init 
41   
42  DBusGMainLoop(set_as_default=True) 
43   
44  _warn(DeprecationWarning("""\ 
45  Importing dbus.glib to use the GLib main loop with dbus-python is deprecated. 
46  Instead, use this sequence: 
47   
48      from dbus.mainloop.glib import DBusGMainLoop 
49   
50      DBusGMainLoop(set_as_default=True) 
51  """), DeprecationWarning, stacklevel=2)
